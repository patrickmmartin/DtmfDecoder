297 procedure TDTMFDecoder.Analyse(const Buffer: PByte; const Size: Cardinal);
298 var
299  i : integer;
300  DataSize : integer;
301  DTMFTones : TDTMFTones;
302  DTMFResult : Char;
303 
304 begin
305 
306  {Quantization is restricted to 8, 16}
307  DataSize := Size div (fQuantisation shr 3);
308 
309  {update chunk time, and account for rollovers }
310  ChunkTime := ChunkTime + round(1000 * DataSize / fFrameRate);
311 
312  for i := 0 to DataSize - 1 do
313  begin
314  case fQuantisation of
315  8:
316  Data[i] := Byte(PByteArray(Buffer)[i]);
317  16:
318  Data[i] := SmallInt(PWordArray(Buffer)[i]);
319  end;
320  end;
321 
322  if Assigned(fOnData) then
323  fOnData(self, PFloat(@Data), DataSize);
324 
325  case fAnalysis of
326  taFFT :
327  DTMFTones := FFTTonePair(DataSize, @Data, fFrameRate );
328  taGoertzel :
329  DTMFTones := GTZTonePair(DataSize, @Data, fFrameRate );
330  end; {case}
331 
332 
333  if Assigned(fOnAnalyse) then
334  fOnAnalyse(self, DTMFTones);
335 
336  DTMFResult := DTMFDigit(DTMFTones);
337 
338  { here is the escape hatch for when the digit processing really needs to be altered
339  beyond what is implemented using this class
340  users can do their own detection of digits based upon the emitted digits, and ignore the OnDetect event}
341 
342  DoDigit(DTMFResult);
343 
344  { else the results of this test could be used in the other event}
345 
346 
347  { desired:
348  a digit is emitted once and only once when the second identical digit
349  arrives after a period of silence
350  ( this eliminates noise very effectively and cheaply )
351  - a silence is deemed to have occurred when fSilenceCount ms have elapsed
352  and no digits have been detected
353  (including those ignored after first emitted digit)
354  }
355 
356 
357  if (DTMFResult <> #0) then
358  begin
359  {digit detected}
360  {if silence has elapsed and the previous digit is same as last detected then emit digit}
361  if ((fSilenceStart <> 0) and ((fSilenceStart + fSilenceCount) < ChunkTime)) and
362  (fLastDetectChar = DTMFResult) then
363  begin
364  fInDigit := true;
365  DoDetect(DTMFResult);
366  fSilenceStart := 0;
367  end;
368  {set last character received marker}
369  fLastCharCount := ChunkTime;
370  end
371  else
372  begin
373  {no digit - decide whether to flag silence start by testing for a sufficiently long dropout}
374  if ((fLastCharCount + fDropoutCount) < ChunkTime) then
375  begin
376  fSilenceStart := fLastCharCount;
377  if (fInDigit) then
378  begin
379  fInDigit := false;
380  DoDetectEnd(fDigit);
381  end;
382  end;
383  end;
384 
385  fLastDetectChar := DTMFResult;
386 
387 
388 end;

390 function TDTMFDecoder.GTZTonePair(NumPoints : integer; Data : PFloatArray ; Rate : single) : TDTMFTones;
391 var
392  i : integer;
393  GState : TNSPSGoertzState;
394  GResults: TSCplx;
395  TonePowers : array of single;
396  SpectrumRecord : PSpectrumRecord;
397  HighRecord : PSpectrumRecord;
398  LowRecord : PSpectrumRecord;
399 
400 begin
401 
402 
403  SetLength(TonePowers, Length(GoertzelFrequencies));
404 
405  {the goertzel analysis is more efficient than an N-sample FFT for m freqs. where m < 2 log2 N
406  this means the breakpoint for simple DTMF ( 8 / 13 bands ) is 512 samples,
407  for which size noise, overhead, appropriateness of the processing and other crud may be more important issues}
408 
409  { run through the set of frequencies }
410  for i := Low(GoertzelFrequencies) to High(GoertzelFrequencies) do
411  begin
412  {uses normalised frequencies}
413  nspsGoertzInit( (GoertzelFrequencies[i] / Rate ) ,GState);
414  GResults := nspsbGoertz(GState, PFloat(Data), NumPoints);
415  with GResults do
416  TonePowers[i] := sqrt(Re*Re + Im*Im) {* 0.01};
417  end;
418 
419  if Assigned(fOnSpectrum) then
420  fOnSpectrum(self, PFloat(@TonePowers[0]), High(GoertzelFrequencies) - Low(GoertzelFrequencies));
421 
422  {choose likeliest tone pair}
423 
424  ClearList(High(GoertzelFrequencies) - Low(GoertzelFrequencies));
425  for i := Low(GoertzelFrequencies) to High(GoertzelFrequencies) do
426  begin
427  SpectrumRecord := ListRecord(i);
428  SpectrumRecord^.Freq := GoertzelFrequencies[i];
429  SpectrumRecord^.Index := i;
430  SpectrumRecord^.Value := TonePowers[i];
431  (* not using emphasis
432  SpectrumRecord^.Value := EmphasisedVal(GoertzelFrequencies[i], TonePowers[i]);
433  *)
434  end;
435 
436  fList.Sort(CompareSpectrum);
437 
438  {just get the top values}
439  HighRecord := PSpectrumRecord(fList[0]);
440  LowRecord := PSpectrumRecord(fList[1]) ;
441 
442 
443  if (HighRecord.Freq < LowRecord.Freq) then
444  begin
445  HighRecord := LowRecord;
446  LowRecord := PSpectrumRecord(fList[0]) ;
447  end;
448 
449  { get average tone pair energy relative to the median value in the set
450  - this seems to be a good test in general}
451 
452  Result.PowerRatio := PeakRatio((Highrecord^.Value + LowRecord^.Value)/ 2, fList);
453 
454  { simple threshhold test to see whether tones are worthwhile }
455  { NB for extremely distorted inputs, testing the average power content is useless,
456  as there is energy content everywhere,
457  a better test is use the absolute value - sad but true}
458 
459  if ( Result.PowerRatio > fPowerRatio ) then
460  begin
461  Result.Status := dsDetected;
462  Result.Low := LowRecord^ ;
463  Result.High := HighRecord^;
464  end
465  else
466  begin
467  Result := NullTone;
468  Result.Status := dsWeak;
469  end
470 
471 
472 end;

474 function TDTMFDecoder.FFTTonePair(NumPoints : integer; Data : PFloatArray ; Rate : single) : TDTMFTones;
475 var
476  i : integer;
477  SpectrumRecord : PSpectrumRecord;
478  HighRecord : TSpectrumRecord;
479  LowRecord : TSpectrumRecord;
480  Frequency : single;
481 begin
482 
483  Result := NullTone;
484  try
485  {optimisation}
486  ClearList(NumPoints - 1);
487 
488  {do transform}
489  nspsDct(PFloat(Data), PFloat(@Spectrum[0]), NumPoints, NSP_DCT_Forward);
490  if Assigned(fOnSpectrum) then
491  fOnSpectrum(self, PFloat(@Spectrum[0]), NumPoints);
492 
493  for i := 0 to NumPoints - 1 do
494  begin
495 
496  Frequency := i * (Rate / ( 2 * NumPoints));
497  if (Frequency > FMinFreq ) and (Frequency < FMaxFreq ) then
498  begin
499  SpectrumRecord := ListRecord(i);
500  SpectrumRecord^.Freq := Frequency;
501  SpectrumRecord^.Value := EmphasisedVal(Frequency, Spectrum[i]);
502  SpectrumRecord^.Index := i;
503  end;
504 
505  end;
506 
507  fList.Sort(CompareSpectrum);
508 
509  {this is based purely upon the numerical values }
510  {a more optimised approach would examine certain frequency bands by value}
511  {but let's keep it general for now}
512 
513  i := 0;
514  {get topmost}
515  HighRecord := PSpectrumRecord(fList[i])^ ;
516  {now seek the next highest that is also beyond the frequency resolution}
517  while (i < NumPoints - 1) and (Abs(HighRecord.Freq - PSpectrumRecord(fList[i])^.Freq) < fFreqResolution) do
518  Inc(i);
519  LowRecord := PSpectrumRecord(fList[i])^;
520 
521  if (HighRecord.Freq < LowRecord.Freq) then
522  begin
523  HighRecord := LowRecord;
524  LowRecord := PSpectrumRecord(fList[0])^ ;
525  end;
526 
527 
528  Result.PowerRatio := PeakRatio((Highrecord.Value + LowRecord.Value)/ 2, fList);
529 
530  {test here for strength}
531 
532  if ( Result.PowerRatio > fPowerRatio ) then
533  begin
534  Result.High := HighRecord;
535  Result.Low := LowRecord;
536  end
537  else
538  begin
539  Result.Status := dsWeak;
540  end;
541  except
542  on E: Exception do
543  begin
544 
545  end;
546  end; {try .. except}
547 
548 end;
